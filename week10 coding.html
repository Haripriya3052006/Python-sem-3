def searchMatrix(m,t):
if not m or not m[0]:
return False
r,c=len(m),len(m[0])
l,r=0,r*c-1
while l<=r:
mid=(l+r)//2
mid1=m[mid//c][mid%c]
if mid1==t:
return True
elif mid1<t:
l=mid+1
else:
r=mid-1
return False
